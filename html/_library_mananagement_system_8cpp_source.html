49 int partition(T * arrList,int start,int end)
50 {
51  T pivot = arrList[end]; // pivot
52  int i = (start - 1); // Index of smaller element
53 
54  for (int j = start; j <= end - 1; j++)
55  {
56  // If current element is smaller than the pivot
57  if (arrList[j] < pivot)
58  {
59  i++; // increment index of smaller element
60  swap(&arrList[i], &arrList[j]);
61  }
62  }
63  swap(&arrList[i + 1], &arrList[end]);
64  return (i + 1);
65  }
